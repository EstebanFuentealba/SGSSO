SELECT * 
FROM TBL_ACTIVIDAD_EVALUADA AEV
	INNER JOIN TBL_DEPARTAMENTO_ORGANIZACION  DO
		ON AEV.ID_DEPARTAMENTO_ORGANIZACION = DO.ID_DEPARTAMENTO_ORGANIZACION
	INNER JOIN TBL_ACTIVIDAD_ESPECIFICA AE ON AEV.ID_ACTIVIDAD_ESPECIFICA = AE.ID_ACTIVIDAD_ESPECIFICA
WHERE DO.ID_ORGANIZACION = @ID_ORGANIZACION
	AND DO.ID_DEPARTAMENTO_ORGANIZACION = @ID_DEPARTAMENTO_ORGANIZACION
	AND AEV.ID_DIVISION = @ID_DIVISION
	AND AEV.ID_AREA = @ID_AREA
	AND AEV.ID_ACTIVIDAD_GENERAL = @ID_ACTIVIDAD_GENERAL
	AND AE.NOM_ACTIVIDAD_ESPECIFICA LIKE '%@NOMBRE_ACTIVIDAD_ESPECIFICA%'
	AND AEV.ID_CARGO = @ID_CARGO
	AND AEV.CONDICION = @CONDICION
	AND AEV.FECHA_CREACION BETWEEN @FECHA_INICIO AND @FECHA_TERMINO
